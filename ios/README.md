Here is how to use build-mac.sh:

./build-mac.sh          # Full build (default)
./build-mac.sh clean    # Clean build directory
./build-mac.sh archive  # Build archive only
./build-mac.sh export   # Export from archive
./build-mac.sh pkg      # Create .pkg from export
./build-mac.sh zip      # Create .zip from export
./build-mac.sh help     # Show help
